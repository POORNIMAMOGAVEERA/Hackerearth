#include<stdio.h>
int main(){
	int t,N;
	scanf("%d",&t);
	while(t--){
	scanf("%d",&N);	
	int arr[N][N],c=0;
    for(int i=0;i<N;i++){
		for(int j=0;j<N;j++){
			scanf("%d",&arr[i][j]);
		}
	}
	for (int i = 0; i < N; i++) {
			for (int j = 0; j < N; j++) {
				for (int x = i; x < N; x++) {
					for (int y = j; y < N; y++) {
						if (arr[x][y] < arr[i][j]) c++;
					}
				}
			}
		}
printf("%d\n",c);
	}
	return 0;
}
